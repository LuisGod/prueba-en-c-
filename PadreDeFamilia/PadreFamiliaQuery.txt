CREATE PROCEDURE PersonajesPorApellido
@pApellido VARCHAR(100)
AS
SET NOCOUNT ON
BEGIN
	SELECT PersonajeID, Nombre, Apellido, Descripcion 
	FROM PadreFamiliaPersonajes 
	WHERE Apellido LIKE @pApellido
END
SET NOCOUNT OFF
GO
